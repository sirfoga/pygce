37 def parse_hh_mm_ss(h):
38  """
39  :param h: str
40  Hours, minutes and seconds in the form hh:mm:ss to parse
41  :return: datetime.time
42  Time parsed
43  """
44 
45  h = str(h).strip() # discard jibberish
46  split_count = h.count(":")

48  return datetime.strptime(str(h).strip(), "%H:%M:%S").time()
49  elif split_count == 1: # mm:ss
50  return datetime.strptime(str(h).strip(), "%M:%S").time()
51  else: # ss
52  return datetime.strptime(str(h).strip(), "%S").time()
