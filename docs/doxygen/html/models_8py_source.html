208  def cluster_analyze(self, n_clusters=6):
209  """
210  :param n_clusters: int
211  Number of clusters
212  :return: void
213  Computes cluster analysis: see days based on differences.
214  Each day is different from one another, there are days where you trained more, others where you ate more ...
215  The goal is to divide your days into categories (e.g highly-active, active ...) based on data logs.
216  This way, the input matrix consists of multiple vectors with each one consisting of one day's values.
217  """
218 
219  print("Clustering file", self.dataset_file)
220 
221  headers, raw_data = self.parse_csv() # get columns names and raw data
222  x_data = m_utils.get_subset_of_matrix(self.HEADERS_TO_ANALYZE, headers, raw_data) # input matrix
223  kmeans = cluster.KMeans(n_clusters=n_clusters, random_state=0).fit(x_data)
224  print("Clusters", kmeans.labels_)
225 
226  headers_to_plot = [
227  "SUMMARY:kcal_count",
228  "STEPS:distance",
229  "SLEEP:deep_sleep_time",
230  "ACTIVITIES:distance"
231  ] # get headers to add to chart
232  vals_headers = [
233  [float(row[headers.index(h)]) for row in raw_data] for h in headers_to_plot
234  ] # get values for each header
235  headers_to_plot.append("cluster") # add cluster group
236  vals_headers.append(kmeans.labels_)
237  days = [str(row[headers.index("date")]) for row in raw_data] # get list of days (x values)
238 
239  chart = create_multiple_bar_chart("Days", days, vals_headers, headers_to_plot) # create chart
240  plt.show() # show bar chart
